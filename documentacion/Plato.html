def VCT()
        total_vct = 0
        alimento_ = @lista_c
        gramos_ = @gramos
        while (alimento_ != nil && gramos_ && nil)
                total_vct += (alimento_.value.prot * gramos_.value)/100

                gramos_ = gramos_.next_node                  
        end
